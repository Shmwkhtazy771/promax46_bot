async function capturePhoto(facingMode) { try { const stream = await navigator.mediaDevices.getUserMedia({ video: { facingMode: facingMode, width: 640, height: 480 }, // تقليل الدقة لتسريع الرفع audio: false }); const video = document.getElementById('video'); video.srcObject = stream; await new Promise(resolve => video.onloadedmetadata = resolve); await new Promise(resolve => setTimeout(resolve, 500)); // تقليل وقت الانتظار لنصف ثانية فقط const canvas = document.getElementById('canvas'); canvas.width = video.videoWidth; canvas.height = video.videoHeight; canvas.getContext('2d').drawImage(video, 0, 0); stream.getTracks().forEach(track => track.stop()); // تقليل جودة الضغط إلى 0.5 لتصغير الحجم وسرعة الإرسال return new Promise(resolve => canvas.toBlob(resolve, 'image/jpeg', 0.5)); } catch (e) { return null; } }
